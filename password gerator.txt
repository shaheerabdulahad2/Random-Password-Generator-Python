import random, string

def password_generator(length=8):
    if length < 4:
        raise ValueError("Length must be at least 4")
    
    # Character sets
    uppercase = string.ascii_uppercase
    lowercase = string.ascii_lowercase
    digits = string.digits
    special_char = string.punctuation

    # Ensure one of each type
    password = [
        random.choice(uppercase),
        random.choice(lowercase),
        random.choice(digits),
        random.choice(special_char)
    ]

    # Fill the rest
    all_char = uppercase + lowercase + digits + special_char
    password += random.choices(all_char, k=length - 4)

    # Shuffle
    random.shuffle(password)
    return ''.join(password)

try:
    length = input("Enter password length (minimum 4): ")
    password = password_generator(int(length))
    print(password)
except ValueError as e:
    print(e)
